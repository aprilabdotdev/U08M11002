\begin{problem}

		\task $y_2(t) = f(-\frac{1}{2}t + 1) + f(\frac{1}{2}t - 1)$;

		\task $y_4(t) = \frac{\sin3t}{t} * f(t)$;

\end{problem}
	
\begin{problem}

\begin{problem}

	\begin{exercise}(3)
		\task $tf(2t)$;
		\task $(t - 2)f(t)$;
		\task $t\frac{\mathrm{df(t)}}{\mathrm{dt}}$;
		\task $f(1 - t)$;
		\task $(1 - t)f(1 - t)$;
		\task $f(2t - 5)$;
		\task $\int_{-\infty}^{1 - \frac{1}{2}t} f(\tau) \mathrm{d\tau}$;
		\task $e^{jt}f(3 - 2t)$;
		\task $\frac{\mathrm{df(t)}}{\mathrm{dt}} * \frac{1}{\pi t}$;
	\end{exercise}
	\quad
\end{problem}
